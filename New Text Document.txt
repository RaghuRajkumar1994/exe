    </select>

    <label for="output-value">Output Quantity:</label>
    <input type="number" id="output-value" placeholder="Enter quantity" required min="1">

    <button onclick="submitOutput()">Submit Output</button>

    <div id="message" class="message"></div>

    <script src="/socket.io/socket.io.js"></script>
    <script>
        // Connect to the Socket.IO server
        const socket = io();
        
        function submitOutput() {
            const workerId = document.getElementById('worker-id').value;
            const outputValue = document.getElementById('output-value').value;
            const messageElement = document.getElementById('message');

            if (outputValue === '' || parseInt(outputValue) <= 0) {
                alert('Please enter a valid quantity.');
                return;
            }

            // Send the data to the server
            socket.emit('submit_output', { 
                workerId: workerId, 
                output: outputValue 
            });

            // Clear the input field and show a confirmation message
            document.getElementById('output-value').value = '';
            messageElement.textContent = 'Output submitted successfully!';
            setTimeout(() => { messageElement.textContent = ''; }, 3000); 
        }
    </script>
